```python
def detect_intent(message):

    if "多少钱" in message:
        return Intent("price", 0.95, 0.8)

    if "考虑一下" in message:
        return Intent("soft_delay", 0.9, 0.4)

    if "不需要" in message:

    return Intent("general", 0.5, 0.3)
```

# 💰 2️⃣ 商业价值评估层（关键）

```python
def evaluate_value(contact, intent):

    base = contact.customer_value

    multiplier = {
        "price": 1.5,
        "negotiation": 1.8,
        "rejection": 0.6,
        "silence_break": 2.0
    }.get(intent.type, 1.0)

    return base * multiplier
```

## 三、数据库完整 Schema

```sql
-- ============================================================
-- WAOS-X v3.0 完整数据库 Schema
-- SQLite WAL Mode + 全表乐观锁 + 审计追踪
-- ============================================================

PRAGMA journal_mode = WAL;
PRAGMA foreign_keys = ON;

-- ==================== 账号层 ====================

-- 微信账号（一个软件管理多个微信号）
CREATE TABLE accounts (
    id              TEXT PRIMARY KEY,           -- UUID
    wx_hash         TEXT NOT NULL UNIQUE,       -- 微信号 SHA256
    wx_nickname     TEXT NOT NULL,              -- 微信昵称
    wx_avatar_url   TEXT,                       -- 头像 URL
    clawbot_token   TEXT,                       -- ClawBot 连接 token
    channel_type    TEXT DEFAULT 'clawbot',     -- clawbot / wecom
    is_active       INTEGER DEFAULT 1,
    daily_msg_count INTEGER DEFAULT 0,          -- 今日消息数（限流用）
    risk_level      INTEGER DEFAULT 0,          -- 风险等级 0-100
    created_at      REAL DEFAULT (unixepoch()),
    updated_at      REAL DEFAULT (unixepoch()),
    version         INTEGER DEFAULT 1
);

-- ==================== 人设层 ====================

-- 人设（每个账号可绑定多个人设，一个人设也可被多个账号共享）
CREATE TABLE personas (
    id              TEXT PRIMARY KEY,           -- UUID
    name            TEXT NOT NULL,              -- 人设名称: "销售专家"
    avatar_emoji    TEXT DEFAULT '🤖',          -- 人设图标
    description     TEXT,                       -- 人设描述

    -- AI 配置
    system_prompt   TEXT NOT NULL,              -- 系统提示词（支持模板变量）
    model_route_id  TEXT,                       -- 关联的模型路由规则
    temperature     REAL DEFAULT 0.7,
    max_tokens      INTEGER DEFAULT 800,
    response_format TEXT DEFAULT 'json',        -- json / text / markdown

    -- 商业规则
    forbidden_rules TEXT DEFAULT '[]',          -- JSON: ["不得承诺降价","不得提及竞品"]
    extract_fields  TEXT DEFAULT '[]',          -- JSON: [{"key":"intent","label":"意向车型","type":"text"}]
    quick_actions   TEXT DEFAULT '[]',          -- JSON: [{"label":"发试驾邀约","template_id":"xxx"}]

    -- 风格控制
    tone            TEXT DEFAULT 'professional',-- professional/casual/warm/aggressive
    max_reply_length INTEGER DEFAULT 120,       -- 最大回复字数
    emoji_style     TEXT DEFAULT 'moderate',    -- none/moderate/rich
    language        TEXT DEFAULT 'zh-CN',

    -- 元数据
    is_default      INTEGER DEFAULT 0,
    is_active       INTEGER DEFAULT 1,
    created_at      REAL DEFAULT (unixepoch()),
    updated_at      REAL DEFAULT (unixepoch()),
    version         INTEGER DEFAULT 1
);

-- 账号 ↔ 人设绑定（多对多）
CREATE TABLE account_personas (
    account_id      TEXT NOT NULL REFERENCES accounts(id),
    persona_id      TEXT NOT NULL REFERENCES personas(id),
    is_primary      INTEGER DEFAULT 0,          -- 是否为该账号的默认人设
    auto_switch_rules TEXT DEFAULT '[]',         -- JSON: 自动建议切换的规则
    created_at      REAL DEFAULT (unixepoch()),
    PRIMARY KEY (account_id, persona_id)
);

-- ==================== 模型路由层 ====================

-- 模型供应商
CREATE TABLE model_providers (
    id              TEXT PRIMARY KEY,
    name            TEXT NOT NULL,              -- "DeepSeek" / "OpenAI" / "Ollama"
    base_url        TEXT NOT NULL,
    api_key_encrypted TEXT,                     -- AES-256 加密存储
    is_local        INTEGER DEFAULT 0,          -- 是否本地模型
    is_active       INTEGER DEFAULT 1,
    created_at      REAL DEFAULT (unixepoch())
);

-- 具体模型
CREATE TABLE models (
    id              TEXT PRIMARY KEY,
    provider_id     TEXT REFERENCES model_providers(id),
    model_name      TEXT NOT NULL,              -- "deepseek-chat" / "gpt-4o"
    display_name    TEXT,                       -- "DeepSeek V3"
    cost_per_1k_input  REAL DEFAULT 0,          -- 输入成本 $/1K tokens
    cost_per_1k_output REAL DEFAULT 0,          -- 输出成本 $/1K tokens
    max_context     INTEGER DEFAULT 8192,       -- 最大上下文窗口
    capabilities    TEXT DEFAULT '[]',          -- JSON: ["text","vision","function_call"]
    avg_latency_ms  INTEGER DEFAULT 2000,
    is_active       INTEGER DEFAULT 1
);

-- 路由规则（决定什么消息走什么模型）
CREATE TABLE model_routes (
    id              TEXT PRIMARY KEY,
    name            TEXT NOT NULL,              -- "高价值客户走GPT-4o"
    priority        INTEGER DEFAULT 0,          -- 规则优先级
    
    -- 匹配条件（JSON）
    match_conditions TEXT NOT NULL,             -- {"account_id":"*", "intent":["price","negotiation"], "customer_value_min": 60}
    
    -- 路由链（按顺序尝试，前者失败走后者）
    model_chain     TEXT NOT NULL,              -- JSON: [{"model_id":"gpt-4o","timeout":8},{"model_id":"deepseek-v3","timeout":12},{"model_id":"qwen-local","timeout":5}]
    
    is_active       INTEGER DEFAULT 1,
    created_at      REAL DEFAULT (unixepoch())
);

-- ==================== 会话与消息层 ====================

-- 会话（一个客户在一个账号下的对话）
CREATE TABLE sessions (
    id              TEXT PRIMARY KEY,
    account_id      TEXT NOT NULL REFERENCES accounts(id),
    persona_id      TEXT REFERENCES personas(id),
    
    -- 客户信息
    contact_name    TEXT,
    contact_wxid    TEXT,                       -- 对方微信 ID（如有）
    contact_avatar  TEXT,
    
    -- 会话状态
    status          TEXT DEFAULT 'active',      -- active/paused/archived
    current_stage   TEXT DEFAULT '新客',        -- CRM 阶段
    last_msg_preview TEXT,
    last_msg_time   REAL DEFAULT 0,
    unread_count    INTEGER DEFAULT 0,
    
    -- 元数据
    tags            TEXT DEFAULT '[]',          -- JSON 标签
    is_vip          INTEGER DEFAULT 0,
    pinned          INTEGER DEFAULT 0,          -- 置顶
    muted           INTEGER DEFAULT 0,          -- 静音（不自动回复）
    
    created_at      REAL DEFAULT (unixepoch()),
    updated_at      REAL DEFAULT (unixepoch()),
    version         INTEGER DEFAULT 1,
    
    UNIQUE(account_id, contact_wxid)
);

CREATE INDEX idx_sessions_account ON sessions(account_id, last_msg_time DESC);
CREATE INDEX idx_sessions_vip ON sessions(is_vip, last_msg_time DESC);

-- 消息
CREATE TABLE messages (
    id              TEXT PRIMARY KEY,           -- UUID v7 (时间有序)
    session_id      TEXT NOT NULL REFERENCES sessions(id),
    account_id      TEXT NOT NULL REFERENCES accounts(id),
    
    seq_id          INTEGER NOT NULL,           -- 会话内消息序号
    role            TEXT NOT NULL,              -- user / assistant / system / tool
    
    -- 内容
    content         TEXT NOT NULL,
    content_hash    TEXT NOT NULL,              -- SHA256(content) 去重
    content_type    TEXT DEFAULT 'text',        -- text/image/link/miniapp/file
```

### 5.2 路由配置代码

```python
class ModelRouter:
    """
    智能模型路由器
    - 按账号/人设/意图/客户价值/成本预算 动态选择模型
    - 自动 Fallback 链
    - 成本追踪与预算控制
    """

    def __init__(self):
        self.daily_budget_usd = 10.0  # 每日预算
        self.today_spent = 0.0

    async def route(self, context: RouteContext) -> ModelChain:
        """
        context 包含:
        - account_id: 哪个微信号
        - persona_id: 当前人设
        - intent: 意图分类结果
        - customer_value: 客户价值评分
        - message_complexity: 消息复杂度 (短/中/长)
        """

        # 1. 预算检查
        if self.today_spent >= self.daily_budget_usd:
            log.warning("今日预算耗尽，强制走本地模型")
            return ModelChain([{"model": "qwen-local", "timeout": 5}])

        # 2. 人设级别路由（最高优先级）
        persona_route = db.fetchone(
            "SELECT model_chain FROM model_routes WHERE persona_id=? AND is_active=1",
            (context.persona_id,)
        )
        if persona_route:
            return ModelChain(json.loads(persona_route["model_chain"]))

        # 3. 智能路由
        if context.customer_value > 80 and context.intent in ("price", "negotiation"):
            return ModelChain([
                {"model": "gpt-4o", "timeout": 10},
                {"model": "deepseek-chat", "timeout": 8},
                {"model": "qwen-local", "timeout": 5},
            ])

        if context.intent in ("complaint", "escalation"):
            return ModelChain([
                {"model": "deepseek-reasoner", "timeout": 15},
                {"model": "deepseek-chat", "timeout": 8},
            ])

        # 4. 默认路由
        return ModelChain([
            {"model": "deepseek-chat", "timeout": 8},
            {"model": "qwen-local", "timeout": 5},
        ])

    async def execute(self, chain: ModelChain, messages: list) -> AIResponse:
        """按链路依次尝试，前者失败走后者"""
        for node in chain.nodes:
            try:
                response = await self.call_model(
                    node["model"], messages, timeout=node["timeout"]
                )
                self.today_spent += response.cost_usd
                return response
            except (TimeoutError, APIError) as e:
                log.warning(f"模型 {node['model']} 失败: {e}，尝试下一个")
                continue

        # 全链路失败：返回安全兜底回复
        return AIResponse(
            reply="好的，我稍后为您详细确认。",
            degraded=True,
            error="全链路失败"
        )
```

### 6.2 实现

```python
class YuanbaoAnalyzer:
    """
    通过元宝进行免费的图片/文档分析
    原理：元宝已置顶在微信对话列表中，通过 ClawBot 可直接发消息给它
    """

    YUANBAO_WXID = "yuanbao_assistant"  # 元宝的微信 ID（需确认）
    YUANBAO_PROMPTS = {
        "moments": "分析这条朋友圈：1.内容类型(生活/工作/广告/情绪) 2.情绪基调 3.是否有商业触发点(提车/升职/搬家等) 4.建议互动策略。返回JSON格式。",
        "video_cover": "分析这个视频号封面和标题，判断：1.内容领域 2.目标人群 3.是否有商业价值 4.适合评论什么。返回JSON。",
        "product_image": "识别这张图片中的产品信息，包括品牌、型号、颜色、大致价格区间。返回JSON。",
        "business_card": "提取名片中的所有信息：姓名、职位、公司、电话、微信、邮箱、地址。返回JSON。",
        "chat_screenshot": "分析这张聊天截图：1.对话双方的关系 2.核心讨论内容 3.情绪走向 4.下一步建议。返回JSON。",
    }

    async def analyze(self, image_path: str, analysis_type: str) -> dict:
        """
        发送图片给元宝，获取分析结果
        """
        prompt = self.YUANBAO_PROMPTS.get(analysis_type, "分析这张图片的内容。返回JSON。")

        # 通过 ClawBot 发送图片+文字给元宝
        response = await clawbot_client.send_media(
            to_wxid=self.YUANBAO_WXID,
            image_path=image_path,
            caption=prompt,
            wait_response=True,
            timeout=30
        )

        # 解析元宝返回的 JSON
        try:
            result = self._extract_json(response.text)
            return {"success": True, "data": result, "source": "yuanbao", "cost": 0}
        except json.JSONDecodeError:
            return {"success": False, "raw": response.text, "source": "yuanbao"}

    def _extract_json(self, text: str) -> dict:
        """从元宝的自然语言回复中提取 JSON"""
        # 尝试直接解析
        try:
            return json.loads(text)
        except:
            pass
        # 尝试提取 ```json ... ``` 代码块
        match = re.search(r'```(?:json)?\s*(\{.*?\})\s*```', text, re.DOTALL)
        if match:
            return json.loads(match.group(1))
        # 尝试提取第一个 {...}
        match = re.search(r'\{.*\}', text, re.DOTALL)
        if match:
            return json.loads(match.group(0))
        raise json.JSONDecodeError("无法提取JSON", text, 0)
```

## 八、关键交互细节（你之前觉得"简陋"的根本原因）

### 8.1 消息发送的完整流程（不是直接发，而是三步确认）

```
用户输入消息 → 回车
       │
       ▼
Step 1: AI 生成回复（显示加载动画：三个跳动的点）
       │
       ▼
Step 2: AI 回复以"草稿卡片"形式出现
       ┌──────────────────────────────────────┐
       │ 🤖 AI 建议回复:                       │
       │                                      │
       │ "李总，M9限量特惠中，本周定还能      │
       │  申请万元补贴。方便来店里感受下？"    │
       │                                      │
       │ ── 策略: 稀缺性+到店邀约 ──           │
       │                                      │
       │ [✅ 直接发送] [✏️ 编辑] [❌ 不用]     │
       └──────────────────────────────────────┘
       │
       ▼
Step 3: 三种选择
  ✅ 直接发送 → ClawBot 发出 → 气泡变绿 → 完成
  ✏️ 编辑 → 变为可编辑文本框 → 修改后发送
  ❌ 不用 → AI 静默，等待人工自己回
```

### 8.2 会话切换的动画

```
点击左侧会话列表的 "张董"
       │
       ▼
1. 当前对话区域向右滑出 (250ms, ease-out)
2. 新对话从左侧滑入 (250ms, ease-in)
3. 右侧 AI 面板内容淡入淡出 (200ms)
4. 顶栏客户信息更新 (渐变)
5. 线索提取表单重置为"待提取..."状态
6. 如果该会话有未读消息，滚动到第一条未读位置
```

### 8.3 通知系统（不是弹窗，是微妙的提示）

```
类型1: 新消息到达 (非当前会话)
  → 左侧会话列表该条目闪一下蓝色
  → 未读计数 +1 (红色圆点)
  → 不打断当前操作

类型2: AI 建议切换人设
  → 顶栏出现浮动条 (从顶部滑下)
  → "💡 检测到客户投诉，建议切换到「客服顾问」人设  [切换] [忽略]"
  → 5秒后自动收起

类型3: 安全拦截
  → 对话区域出现红色边框卡片
  → 轻微震动动画 (水平位移 ±3px, 2次)
  → 风控面板角标 +1

类型4: 元宝分析完成
  → 朋友圈面板出现绿色脉冲点
  → "✨ 元宝分析完成: 张董晒了新车 → 商业触发器!"

类型5: 每日战报
  → 每天 18:00 在看板页生成
  → 顶栏 📊 图标出现蓝色脉冲
  → 点击展开战报卡片
```

### 8.4 快捷键系统

```yaml
客户管理:
```

## 九、技术选型（2026年最优解）

```yaml
前端框架:    PyQt6 + QML (不是 QWidget, 是 QML 声明式 UI)
              理由: QML 原生支持动画、状态机、粒子效果
              配合 Fluid 组件库实现苹果级视觉

后端核心:    Python 3.12+ (asyncio)
              理由: 与 ClawBot SDK 同生态

消息通道:    微信 ClawBot (iLink 协议, 长轮询)
              备选: 企业微信 API

AI 编排:     OpenClaw MCP Server
              理由: 官方生态, 工具可插拔, 未来扩展性

数据库:      SQLite WAL (本地) + Redis Streams (可选, 多进程场景)

向量检索:    ChromaDB (本地嵌入, bge-m3)
              用于 RAG 知识库 + 记忆系统

图片分析:    元宝 (免费, 微信内) + Qwen-VL (本地, 需要时)

打包部署:    PyInstaller + NSIS 安装包
              或: Electron 壳 + Python 后端 (如果未来要做 Web)

自动更新:    自建更新服务器 + 增量更新
```

## 十一、你之前简陋的根源——总结

| 简陋感来源 | 旧做法 | 新做法 |
|-----------|--------|--------|
| UI 无层次 | 所有元素平铺 | Focus Mode + 三栏聚焦 + 抽屉展开 |
| 无动效 | 状态直接跳变 | 滑动/淡入/脉冲/震动，每个交互有反馈 |
| 按钮无意义 | 开关只改状态 | 每个按钮有前置确认和后置反馈 |
| 空状态丑 | "暂无数据" | 引导性插画 + "点击创建第一个..." |
| 错误冷冰冰 | QMessageBox弹窗 | Toast + 内联错误 + 自动修复建议 |
| 信息密度低 | 一个页面只做一件事 | 看板聚合多维度，但分层展示 |
| 无搜索 | 列表只能肉眼看 | Ctrl+K 全局搜索，模糊匹配+高亮 |
| 无快捷键 | 只能鼠标点 | 20+ 快捷键覆盖所有高频操作 |
| 颜色无体系 | 随机用色 | 6色设计系统，每个颜色有语义 |
| 数据无故事 | 只显示数字 | 趋势图+对比+百分比+建议 |
