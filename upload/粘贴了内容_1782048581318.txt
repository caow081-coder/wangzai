class NodeType(Enum):
    START = "start"
    SKILL = "skill"
    CONDITION = "condition"
    WAIT = "wait"
    NOTIFY = "notify"
    END = "end"


@dataclass
class SOPNode:
    id: str
    type: NodeType
    label: str
    skill_id: Optional[str] = None
    config: Dict[str, Any] = field(default_factory=dict)
    position_x: float = 0.0
    position_y: float = 0.0


@dataclass
class SOPEdge:
    id: str
    source: str
    target: str
    condition: Optional[str] = None  # 条件节点的分支标签 (YES/NO)


@dataclass
class SOPDefinition:
    id: str
    name: str
    description: str
    trigger_type: str  # manual / auto_schedule / auto_event
    trigger_condition: Dict[str, Any]  # JSON
    nodes: List[SOPNode]
    edges: List[SOPEdge]
    is_active: bool = True
    created_at: float = 0.0
    updated_at: float = 0.0


@dataclass
class SOPInstance:
    id: str
    sop_id: str
    customer_id: str
    current_node_id: str
    status: str  # running / paused / completed / failed / aborted
    context: Dict[str, Any]
    started_at: float
    completed_at: Optional[float] = None
    created_at: float = 0.0
    updated_at: float = 0.0


@dataclass
class SOPNodeLog:
    instance_id: str
    node_id: str
    skill_name: str
    input_data: Dict[str, Any]
    output_data: Dict[str, Any]
    status: str  # success / failed / skipped
    error_message: Optional[str] = None
    started_at: float = 0.0
    completed_at: float = 0.0
    duration_ms: int = 0

class SOPStorage:
    """SOP 数据存储 (SQLite)"""
    def __init__(self, db_conn):
        self.conn = db_conn
        self._init_tables()

    def _init_tables(self):
        # SOP 定义表
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS sop_definitions (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL,
                description TEXT,
                trigger_type TEXT,
                trigger_condition TEXT,
                nodes TEXT NOT NULL,
                edges TEXT NOT NULL,
                is_active INTEGER DEFAULT 1,
                created_at REAL DEFAULT (unixepoch()),
                updated_at REAL DEFAULT (unixepoch()),
                version INTEGER DEFAULT 1
            )
        """)
        # SOP 执行实例表
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS sop_instances (
                id TEXT PRIMARY KEY,
                sop_definition_id TEXT NOT NULL,
                customer_id TEXT NOT NULL,
                current_node_id TEXT,
                status TEXT,
                context TEXT,
                started_at REAL,
                completed_at REAL,
                created_at REAL DEFAULT (unixepoch()),
                updated_at REAL DEFAULT (unixepoch())
            )
        """)
        # SOP 节点日志表
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS sop_node_logs (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                sop_instance_id TEXT NOT NULL,
                node_id TEXT NOT NULL,
                skill_name TEXT,
                input TEXT,
                output TEXT,
                status TEXT,
                error_message TEXT,
                started_at REAL,
                completed_at REAL,
                duration_ms INTEGER
            )
        """)
        # Skill 注册表 (UI 工具箱数据源)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS skill_registry (
                id TEXT PRIMARY KEY,
                name TEXT NOT NULL,
                description TEXT,
                category TEXT,
                input_schema TEXT,
                output_schema TEXT,
                is_active INTEGER DEFAULT 1
            )
        """)
        self.conn.commit()
        self._seed_skills()

    def _seed_skills(self):
        """从 SkillRegistry 同步到 skill_registry 表 (UI 用)"""
        registry = SkillRegistry()
        for skill in registry.list_all():
            self.conn.execute(
                "INSERT OR REPLACE INTO skill_registry (id, name, description, category, is_active) VALUES (?, ?, ?, ?, ?)",
                (skill.id, skill.name, skill.description, skill.category.value, 1)
            )
        self.conn.commit()

    def save_definition(self, sop: SOPDefinition):
        self.conn.execute(
            "INSERT OR REPLACE INTO sop_definitions (id, name, description, trigger_type, trigger_condition, nodes, edges, is_active, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
            (
                sop.id, sop.name, sop.description,
                sop.trigger_type,
                json.dumps(sop.trigger_condition),
                json.dumps([{"id": n.id, "type": n.type.value, "label": n.label, "skill_id": n.skill_id, "config": n.config, "x": n.position_x, "y": n.position_y} for n in sop.nodes]),
                json.dumps([{"id": e.id, "source": e.source, "target": e.target, "condition": e.condition} for e in sop.edges]),
                1 if sop.is_active else 0,
                time.time()
            )
        )
        self.conn.commit()

    def get_definition(self, sop_id: str) -> Optional[SOPDefinition]:
        row = self.conn.execute("SELECT * FROM sop_definitions WHERE id=?", (sop_id,)).fetchone()
        if not row:
            return None
        nodes_data = json.loads(row["nodes"])
        edges_data = json.loads(row["edges"])
        return SOPDefinition(
            id=row["id"],
            name=row["name"],
            description=row["description"],
            trigger_type=row["trigger_type"],
            trigger_condition=json.loads(row["trigger_condition"]) if row["trigger_condition"] else {},
            nodes=[SOPNode(
                id=n["id"],
                type=NodeType(n["type"]),
                label=n["label"],
                skill_id=n.get("skill_id"),
                config=n.get("config", {}),
                position_x=n.get("x", 0.0),
                position_y=n.get("y", 0.0)
            ) for n in nodes_data],
            edges=[SOPEdge(id=e["id"], source=e["source"], target=e["target"], condition=e.get("condition")) for e in edges_data],
            is_active=bool(row["is_active"]),
            created_at=row["created_at"],
            updated_at=row["updated_at"]
        )

    def list_definitions(self) -> List[Dict]:
        rows = self.conn.execute("SELECT id, name, description, trigger_type, is_active, created_at FROM sop_definitions").fetchall()
        return [dict(row) for row in rows]

    def create_instance(self, instance: SOPInstance):
        self.conn.execute(
            "INSERT INTO sop_instances (id, sop_definition_id, customer_id, current_node_id, status, context, started_at) VALUES (?, ?, ?, ?, ?, ?, ?)",
            (instance.id, instance.sop_id, instance.customer_id, instance.current_node_id, instance.status, json.dumps(instance.context), instance.started_at)
        )
        self.conn.commit()

    def update_instance(self, instance: SOPInstance):
        self.conn.execute(
            "UPDATE sop_instances SET current_node_id=?, status=?, context=?, completed_at=?, updated_at=? WHERE id=?",
            (instance.current_node_id, instance.status, json.dumps(instance.context), instance.completed_at, time.time(), instance.id)
        )
        self.conn.commit()

    def get_instance(self, instance_id: str) -> Optional[SOPInstance]:
        row = self.conn.execute("SELECT * FROM sop_instances WHERE id=?", (instance_id,)).fetchone()
        if not row:
            return None
        return SOPInstance(
            id=row["id"],
            sop_id=row["sop_definition_id"],
            customer_id=row["customer_id"],
            current_node_id=row["current_node_id"],
            status=row["status"],
            context=json.loads(row["context"]) if row["context"] else {},
            started_at=row["started_at"],
            completed_at=row["completed_at"],
            created_at=row["created_at"],
            updated_at=row["updated_at"]
        )

    def log_node(self, log: SOPNodeLog):
        self.conn.execute(
            "INSERT INTO sop_node_logs (sop_instance_id, node_id, skill_name, input, output, status, error_message, started_at, completed_at, duration_ms) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",
            (log.instance_id, log.node_id, log.skill_name, json.dumps(log.input_data), json.dumps(log.output_data), log.status, log.error_message, log.started_at, log.completed_at, log.duration_ms)
        )
        self.conn.commit()

    def get_instance_logs(self, instance_id: str) -> List[Dict]:
        rows = self.conn.execute("SELECT * FROM sop_node_logs WHERE sop_instance_id=? ORDER BY id", (instance_id,)).fetchall()
        return [dict(row) for row in rows]

# 5. SOP 运行时引擎
# ==========================================

class SOPRuntime:
    """SOP 执行引擎 - 核心"""
    def __init__(self, storage: SOPStorage):
        self.storage = storage
        self._running_instances: Dict[str, SOPInstance] = {}
        self._registry = SkillRegistry()

    def start_instance(self, sop_id: str, customer_id: str, initial_context: Dict = None) -> str:
        """启动一个新的 SOP 实例"""
        definition = self.storage.get_definition(sop_id)
        if not definition:
            raise ValueError(f"SOP definition not found: {sop_id}")

        # 找第一个 START 节点
        start_node = next((n for n in definition.nodes if n.type == NodeType.START), None)
        if not start_node:
            raise ValueError("SOP has no START node")

        instance_id = str(uuid.uuid4())
        instance = SOPInstance(
            id=instance_id,
            sop_id=sop_id,
            customer_id=customer_id,
            current_node_id=start_node.id,
            status="running",
            context=initial_context or {},
            started_at=time.time()
        )
        self.storage.create_instance(instance)
        self._running_instances[instance_id] = instance

        # 异步执行
        thread = threading.Thread(target=self._execute, args=(instance_id,), daemon=True)
        thread.start()

        return instance_id

    def _execute(self, instance_id: str):
        """执行 SOP 的主循环"""
        instance = self._running_instances.get(instance_id)
        if not instance:
            return

        definition = self.storage.get_definition(instance.sop_id)
        if not definition:
            return

        # 构建节点映射
        nodes_map = {n.id: n for n in definition.nodes}
        edges_map = {}  # source -> list of (target, condition)
        for e in definition.edges:
            if e.source not in edges_map:
                edges_map[e.source] = []
            edges_map[e.source].append((e.target, e.condition))

        current_node_id = instance.current_node_id

        while current_node_id and instance.status == "running":
            node = nodes_map.get(current_node_id)
            if not node:
                break

            # 执行当前节点
            result = self._execute_node(instance, node, definition)

            if result["status"] == "failed":
                instance.status = "failed"
                self.storage.update_instance(instance)
                break

            if result["status"] == "paused":
                instance.status = "paused"
                self.storage.update_instance(instance)
                break

            # 找下一个节点
            edges = edges_map.get(current_node_id, [])
            next_node_id = None
            if len(edges) == 1:
                next_node_id = edges[0][0]
            elif len(edges) > 1:
                # 条件分支
                for target, condition in edges:
                    if condition and result.get("branch") == condition:
                        next_node_id = target
                        break
                    elif not condition:  # 默认分支
                        next_node_id = target

            current_node_id = next_node_id
            instance.current_node_id = current_node_id
            self.storage.update_instance(instance)

            # 如果到达 END 节点
            if current_node_id and nodes_map.get(current_node_id, NodeType.START).type == NodeType.END:
                instance.status = "completed"
                instance.completed_at = time.time()
                self.storage.update_instance(instance)
                break

    def _execute_node(self, instance: SOPInstance, node: SOPNode, definition: SOPDefinition) -> Dict:
        """执行单个节点"""
        start_time = time.time()
        log = SOPNodeLog(
            instance_id=instance.id,
            node_id=node.id,
            skill_name="",
            input_data={},
            output_data={},
            status="success",
            started_at=start_time
        )

        try:
            if node.type == NodeType.SKILL:
                # 执行 Skill
                skill = self._registry.get(node.skill_id)
                if not skill:
                    raise ValueError(f"Skill not found: {node.skill_id}")

                # 合并上下文
                ctx = instance.context.copy()
                ctx.update(node.config)

                result = self._registry.execute(node.skill_id, ctx)
                log.skill_name = skill.name
                log.input_data = ctx
                log.output_data = result

                # 将结果合并到实例上下文
                instance.context.update(result)

            elif node.type == NodeType.CONDITION:
                # 条件判断
                field = node.config.get("field", "")
                operator = node.config.get("operator", ">=")
                value = node.config.get("value", 0)

                actual = instance.context.get(field, 0)
                if operator == ">=":
                    result = {"branch": "YES" if actual >= value else "NO"}
                elif operator == ">":
                    result = {"branch": "YES" if actual > value else "NO"}
                elif operator == "==":
                    result = {"branch": "YES" if actual == value else "NO"}
                else:
                    result = {"branch": "NO"}

                log.output_data = result

            elif node.type == NodeType.WAIT:
                # 延迟等待
                seconds = node.config.get("seconds", 60)
                time.sleep(seconds)
                log.output_data = {"waited": seconds}

            elif node.type == NodeType.NOTIFY:
                # 通知人工
                message = node.config.get("message", "SOP 节点执行完成")
                log.output_data = {"notified": True, "message": message}
                # 发送桌面通知 (通过信号)
                if hasattr(self, "bus"):
                    self.bus.notify.emit(message)

            elif node.type == NodeType.START:
                log.output_data = {"started": True}

            elif node.type == NodeType.END:
                log.output_data = {"ended": True}

            log.status = "success"

        except Exception as e:
            log.status = "failed"
            log.error_message = str(e)
            log.output_data = {"error": str(e)}

        log.completed_at = time.time()
        log.duration_ms = int((log.completed_at - log.started_at) * 1000)

        self.storage.log_node(log)
        return {"status": log.status, "branch": log.output_data.get("branch") if node.type == NodeType.CONDITION else None}

    def pause(self, instance_id: str):
        instance = self._running_instances.get(instance_id)
        if instance:
            instance.status = "paused"
            self.storage.update_instance(instance)

    def resume(self, instance_id: str):
        instance = self._running_instances.get(instance_id)
        if instance and instance.status == "paused":
            instance.status = "running"
            self.storage.update_instance(instance)
            # 重新触发执行
            thread = threading.Thread(target=self._execute, args=(instance_id,), daemon=True)
            thread.start()

    def stop(self, instance_id: str):
        instance = self._running_instances.get(instance_id)
        if instance:
            instance.status = "aborted"
            instance.completed_at = time.time()
            self.storage.update_instance(instance)
            del self._running_instances[instance_id]

    def get_status(self, instance_id: str) -> Optional[Dict]:
        instance = self.storage.get_instance(instance_id)
        if not instance:
            return None
        logs = self.storage.get_instance_logs(instance_id)
        return {

    def _load_default(self):
        """加载第一个SOP或创建默认"""
        sops = self.storage.list_definitions()
        if sops:
            self.load_sop(sops[0]["id"])
        else:
            default = self._create_default_sop()
            self.storage.save_definition(default)
            self.load_sop(default.id)

    def load_sop(self, sop_id: str):
        self.current_sop_id = sop_id
        definition = self.storage.get_definition(sop_id)
        if not definition:
            return
        self.nodes = definition.nodes
        self.edges = definition.edges
        self._render()

    def _render(self):
        """渲染画布"""
        text = f"📋 SOP: {self.current_sop_id[:8]}\n"
        text += "━" * 40 + "\n\n"

        # 绘制节点
        for node in self.nodes:
            icon = {
                NodeType.START: "🟢",
                NodeType.SKILL: "🔵",
                NodeType.CONDITION: "🔶",
                NodeType.WAIT: "⏳",
                NodeType.NOTIFY: "🔔",
                NodeType.END: "🟥"
            }.get(node.type, "⬜")

            skill_name = ""
            if node.skill_id:
                skill = SkillRegistry().get(node.skill_id)
                if skill:
                    skill_name = f" [{skill.name}]"

            label = f"{icon} {node.label}{skill_name}"
            if self.selected_node and self.selected_node.id == node.id:
                label = f">>> {label} <<<"
            text += label + "\n"

        # 绘制边
        text += "\n" + "━" * 40 + "\n"
        for edge in self.edges:
            cond = f" [{edge.condition}]" if edge.condition else ""
            text += f"  {edge.source} → {edge.target}{cond}\n"

        # 统计
        text += "\n" + "━" * 40 + "\n"
        text += f"📊 节点: {len(self.nodes)}  |  连线: {len(self.edges)}"

        self.canvas.setText(text)

    def _add_skill_node(self):
        node_id = f"skill_{len(self.nodes)+1}"
        node = SOPNode(
            id=node_id,
            type=NodeType.SKILL,
            label=f"技能节点 {len(self.nodes)+1}",
            skill_id="reply_generate",
            config={},
            position_x=100,
            position_y=len(self.nodes) * 60 + 50
        )
        self.nodes.append(node)
        self._render()
        self.selected_node = node
        self._update_property_panel()

    def _add_condition_node(self):
        node_id = f"cond_{len(self.nodes)+1}"
        node = SOPNode(
            id=node_id,
            type=NodeType.CONDITION,
            label=f"条件判断 {len(self.nodes)+1}",
            config={"field": "value", "operator": ">=", "value": 80},
            position_x=100,
            position_y=len(self.nodes) * 60 + 50
        )
        self.nodes.append(node)
        self._render()

    def _add_wait_node(self):
        node_id = f"wait_{len(self.nodes)+1}"
        node = SOPNode(
            id=node_id,
            type=NodeType.WAIT,
            label=f"等待节点 {len(self.nodes)+1}",
            config={"seconds": 60},
            position_x=100,
            position_y=len(self.nodes) * 60 + 50
        )
        self.nodes.append(node)
        self._render()

    def _update_property_panel(self):
        if not self.selected_node:
            return
        self.prop_name.setText(self.selected_node.label)
        if self.selected_node.skill_id:
            idx = self.prop_skill.findData(self.selected_node.skill_id)
            if idx >= 0:
                self.prop_skill.setCurrentIndex(idx)
        self.prop_config.setText(json.dumps(self.selected_node.config, indent=2, ensure_ascii=False))

    def _delete_selected(self):
        if not self.selected_node:
            return
        self.nodes = [n for n in self.nodes if n.id != self.selected_node.id]
        self.edges = [e for e in self.edges if e.source != self.selected_node.id and e.target != self.selected_node.id]
        self.selected_node = None
        self._render()

    def _save(self):
        if not self.current_sop_id:
            return

        definition = self.storage.get_definition(self.current_sop_id)
        if definition:
            definition.nodes = self.nodes
            definition.edges = self.edges
            definition.updated_at = time.time()
            self.storage.save_definition(definition)
            QMessageBox.information(self, "保存成功", f"SOP {definition.name} 已保存")

class SOPWorkbenchIntegration:
    """将 SOP 引擎集成到现有工作台"""
    def __init__(self, runtime: SOPRuntime, storage: SOPStorage):
        self.runtime = runtime
        self.storage = storage

    def add_sop_trigger(self, customer_id: str, message: str, base_value: float) -> Optional[str]:
        """检测是否有匹配的 SOP 并触发"""
        # 先检测意图
        from app_waos_x_final import WAOSXEngine
        intent = WAOSXEngine.detect_intent(message)

        # 查找匹配的 SOP
        sops = self.storage.list_definitions()
        for sop_info in sops:
            definition = self.storage.get_definition(sop_info["id"])
            if not definition or not definition.is_active:
                continue

            # 检查触发条件
            trigger = definition.trigger_condition
            if trigger.get("intent") == intent.type:
                min_value = trigger.get("min_value", 0)
                if base_value >= min_value:
                    # 启动 SOP
                    return self.runtime.start_instance(definition.id, customer_id, {
                        "message": message,
                        "intent": intent.type,
                        "base_value": base_value,
                        "value": WAOSXEngine.evaluate_value(base_value, intent)
                    })
        return None
