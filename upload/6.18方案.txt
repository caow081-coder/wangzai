# Redis分布式锁

lock_manager.py

```python
from redis.asyncio import Redis

redis = Redis()

class LockManager:

    async def lock(
        self,
        account_id:str
    ):

        return await redis.set(
            f"lock:{account_id}",
            "1",
            ex=30,
            nx=True
        )

    async def unlock(
        self,
        account_id:str
    ):

        await redis.delete(
            f"lock:{account_id}"
        )
```

# ② EventBus（核心中枢）

```python id="event_bus"
import asyncio
from collections import defaultdict

class EventBus:

    def __init__(self):
        self.subscribers = defaultdict(list)

    def subscribe(self, event_type, handler):
        self.subscribers[event_type].append(handler)

    async def publish(self, event_type, data):
        if event_type not in self.subscribers:
            return

        tasks = [
            handler(data)
            for handler in self.subscribers[event_type]
        ]

        await asyncio.gather(*tasks)
```

```python id="main"
from fastapi import FastAPI, WebSocket
from ws import manager
from persona import PersonaEngine
from scoring import score_customer

app = FastAPI()

engine = PersonaEngine()

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await manager.connect(websocket)

    try:
        while True:
            data = await websocket.receive_json()

            msg = data["message"]
            customer = data["customer"]

            # 1. 客户评分
            score = score_customer(msg, customer)

            # 2. Persona处理
            result = engine.run(msg, score)

            # 3. 返回前端
            await manager.send({
                "reply": result["reply"],
                "score": score,
                "action": result.get("action")
            })

    except:
        manager.disconnect(websocket)
```

## 3️⃣ ws.py（WebSocket管理）

```python id="ws"
class ConnectionManager:

    def __init__(self):
        self.active = []

    async def connect(self, ws):
        await ws.accept()
        self.active.append(ws)

    def disconnect(self, ws):
        self.active.remove(ws)

    async def send(self, msg):
        for ws in self.active:
            await ws.send_json(msg)

manager = ConnectionManager()
```

```python id="score"
def score_customer(message, customer):

    score = 50

    # 关键词意图
    if "多少钱" in message:
        score += 20

    if "考虑" in message:
        score += 15

    if "再看看" in message:
        score -= 10

    # 客户历史
    if customer.get("history") == "active":
        score += 15

    if customer.get("purchased"):
        score += 30

    # 上限
    return min(100, max(0, score))
```

## 📦 代码核心

```python id="ai_brain"
class AISalesBrain:

    def __init__(self, llm, memory, strategy_engine):
        self.llm = llm
        self.memory = memory
        self.strategy = strategy_engine

    async def run(self, message, customer_id):

        # 1. 记忆召回
        memories = await self.memory.search(customer_id, message)

        # 2. 策略选择
        strategy = self.strategy.select(customer_id, memories)

        # 3. LLM决策
        prompt = self.build_prompt(message, memories, strategy)

        reply = await self.llm.generate(prompt)

        # 4. 行动规划
        action = self.plan_action(reply, strategy)

        return {
            "reply": reply,
            "action": action,
            "strategy": strategy
        }
```

## 1️⃣ AI Brain 主入口

```python id="ai_brain_main"
class AIBrain:

    def __init__(self, llm, memory, crm, persona_engine):

        self.llm = llm
        self.memory = memory
        self.crm = crm
        self.persona = persona_engine

    async def run(self, message, customer_id):

        # 1. 客户画像
        profile = await self.crm.get_profile(customer_id)

        # 2. 记忆召回
        memories = await self.memory.search(customer_id, message)

        # 3. Persona选择
        persona = self.persona.select(profile)

        # 4. 策略生成
        strategy = persona.strategy(profile, memories)

        # 5. LLM生成
        prompt = self.build_prompt(message, profile, memories, strategy)

        reply = await self.llm.generate(prompt)

        # 6. 行动规划
        action = self.plan(reply, strategy)

        return {
            "reply": reply,
            "action": action,
            "persona": persona.name
        }
```

# 🧠 四、Event Bus（系统核心）

```python id="bus"
class EventBus:

    def __init__(self):
        self.subscribers = []

    def subscribe(self, fn):
        self.subscribers.append(fn)

    def publish(self, event):

        for fn in self.subscribers:
            fn(event)
```

## ✔ 修复方案：统一状态中心（UserState Kernel）

```python id="state_kernel"
class UserStateKernel:

    def __init__(self):
        self.state = {}

    def get(self, user_id):
        return self.state.get(user_id, {
            "score": 0,
            "stage": "cold",
            "persona": "FRIEND"
        })

    def update(self, user_id, patch: dict):

        if user_id not in self.state:
            self.state[user_id] = self.get(user_id)

        for k, v in patch.items():
            self.state[user_id][k] = v

        return self.state[user_id]
```
