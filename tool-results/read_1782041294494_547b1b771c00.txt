   319→// ─── 压测监控面板 ────────────────────────────────────────────

   327→  const intervalMin = Math.floor(sm.intervalMs / 60000)
   328→  const lastRoundAgo = sm.lastRoundAt ? Math.floor((Date.now() - sm.lastRoundAt) / 1000) : 0
   329→

   331→    <div className="shrink-0 border-b border-border/60 bg-card">

   335→        <span className="text-[11px] font-semibold">压测监控</span>

   338→            第{sm.currentRound}轮 · {durationMin}分钟 · 每{intervalMin}分钟

   848→function timeAgo(iso: string): string {
   849→  const diff = Date.now() - new Date(iso).getTime()
   850→  if (diff < 60000) return '刚刚'
   851→  if (diff < 3600000) return `${Math.floor(diff / 60000)}m`
   852→  if (diff < 86400000) return `${Math.floor(diff / 3600000)}h`
   853→  return `${Math.floor(diff / 86400000)}d`
   854→}
