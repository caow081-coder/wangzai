DATA_DIR = os.path.join(os.path.expanduser("~"), ".waos_x")
os.makedirs(DATA_DIR, exist_ok=True)

class Database:
    def __init__(self):
        self.conn = sqlite3.connect(os.path.join(DATA_DIR, "core.db"), check_same_thread=False)
        self.conn.row_factory = sqlite3.Row
        self._w_lock = threading.Lock()
        self.conn.execute("PRAGMA journal_mode=WAL;")
        self.conn.execute("""CREATE TABLE IF NOT EXISTS messages (
            id INTEGER PRIMARY KEY AUTOINCREMENT, wx_id TEXT, role TEXT, content TEXT, timestamp REAL
        )""")
        self.conn.execute("""CREATE TABLE IF NOT EXISTS crm_leads (
            id TEXT PRIMARY KEY, name TEXT, intent TEXT, value REAL, status TEXT, version INTEGER DEFAULT 1
        )""")
        self._init_dummy_data()
        self.conn.commit()

    def _init_dummy_data(self):
        self.conn.execute("DELETE FROM crm_leads")
        leads = [
            ("u001", "李总", "M9", 85.0, "新客", 1),
            ("u002", "张董", "L9", 60.0, "已报价", 1),
            ("u003", "王哥", "S9", 40.0, "已邀约", 2)
        ]
        for l in leads: self.conn.execute("INSERT INTO crm_leads VALUES (?,?,?,?,?,?)", l)

    def execute(self, sql, args=()):
        with self._w_lock:
            cur = self.conn.cursor(); cur.execute(sql, args); self.conn.commit()
            return cur

    def fetchone(self, sql, args=()):
        return self.conn.execute(sql, args).fetchone()

    def fetchall(self, sql, args=()):
        return self.conn.execute(sql, args).fetchall()

db = Database()

@dataclass
class Intent:
    type: str
    confidence: float
    urgency: float

class WAOSXEngine:
    @staticmethod
    def detect_intent(message: str) -> Intent:
        if "多少钱" in message or "价格" in message: return Intent("price", 0.95, 0.8)
        if "不需要" in message or "别烦" in message: return Intent("rejection", 0.85, 0.7)
        if "在吗" in message or "最近" in message: return Intent("silence_break", 0.9, 0.9)
        return Intent("general", 0.5, 0.3)

    @staticmethod
    def evaluate_value(base_value: float, intent: Intent) -> float:
        multiplier = {"price": 1.5, "rejection": 0.6, "silence_break": 2.0}.get(intent.type, 1.0)
        return base_value * multiplier

    @staticmethod
    def select_strategy(value: float, intent: Intent) -> str:
        if value > 80 and intent.type in ["price", "silence_break"]: return "close_now"
        if intent.type == "rejection": return "soft_recovery"
        if intent.type == "silence_break": return "reconnect_hook"
        return "standard_reply"

    @staticmethod
    def generate_reply(strategy: str, msg: str) -> dict:
        """模拟大模型生成话术与提取字段"""
        time.sleep(random.uniform(1.0, 2.0)) # 模拟思考延迟
        if strategy == "close_now":
            return {"reply": f"李总，M9现车不多，今天定的话我申请内部试驾名额和万元补贴，您下午方便来吗？", "extracted": {"意向": "M9", "预算": "50万"}, "status": "高意向"}
        elif strategy == "soft_recovery":
            return {"reply": "理解的李总，买车确实得慎重。我不催您，这是最新的电车评测报告您留着参考，有疑问随时找我。", "extracted": {"情绪": "抗拒"}, "status": "跟进中"}
        elif strategy == "reconnect_hook":
            return {"reply": "李总好久不见！上次听您说家里两个娃，最近新出了个儿童安全座椅评测，感觉特别适合您，发给您看看？", "extracted": {"家庭": "两孩"}, "status": "重新激活"}
        return {"reply": "收到，我马上发您配置单。", "extracted": {}, "status": "新客"}

class Inspector:
    DANGER_PATTERN = re.compile(r"(降价|便宜|保证|送|最低价)")
    
    @staticmethod
    def check_takeover(wx_id: str) -> bool:
        """防双端打架：查库必须在写入前"""
        last_msg = db.fetchone("SELECT role, timestamp FROM messages WHERE wx_id=? ORDER BY timestamp DESC LIMIT 1", (wx_id,))
        if last_msg and last_msg['role'] == 'assistant' and (time.time() - last_msg['timestamp'] < 10):
            return True # 拦截
        return False

    @staticmethod
    def check_danger(text: str) -> bool:
        if Inspector.DANGER_PATTERN.search(text): return False
        return True

# 6. 业务处理线程 (全链路闭环)
# ==========================================
class ProcessWorker(QThread):
    def __init__(self, wx_id: str, msg: str, base_value: float):
        super().__init__()
        self.wx_id = wx_id
        self.msg = msg
        self.base_value = base_value

    def run(self):
        try:
            bus.status_update.emit("WAOS-X 决策中...", "#f9e2af")
            bus.new_bubble.emit(self.msg, True, False)

            # 1. 防双端打架 (修复时序)
            if Inspector.check_takeover(self.wx_id):
                bus.show_takeover.emit(True)
                bus.new_bubble.emit("[系统拦截] 防双端打架触发，AI 沉默", False, True)
                bus.status_update.emit("防打架拦截", "#f38ba8")
                return

            # 2. WAOS-X 引擎运转
            intent = WAOSXEngine.detect_intent(self.msg)
            value = WAOSXEngine.evaluate_value(self.base_value, intent)
            strategy = WAOSXEngine.select_strategy(value, intent)
            bus.log_msg.emit(f"[WAOS-X] 意图:{intent.type} | 价值:{value:.1f} | 策略:{strategy}", "info")

            # 3. 生成回复
            result = WAOSXEngine.generate_reply(strategy, self.msg)
            reply = result.get("reply", "解析失败")
            extracted = result.get("extracted", {})
            new_status = result.get("status", "新客")

            # 4. 高危熔断
            if not Inspector.check_danger(reply):
                bus.new_bubble.emit(reply, False, True)
                bus.status_update.emit("熔断拦截", "#f38ba8")
                return

            # 5. 写入数据库 (先查后写，防双端打架安全)
            db.execute("INSERT INTO messages (wx_id, role, content, timestamp) VALUES (?, ?, ?, ?)", 
                       (self.wx_id, "user", self.msg, time.time()))
            
            # 模拟打字延迟
            bus.status_update.emit("拟人打字中...", "#89dceb")
            self.msleep(int(random.uniform(1000, 2000)))
            
            bus.new_bubble.emit(reply, False, False)
            db.execute("INSERT INTO messages (wx_id, role, content, timestamp) VALUES (?, ?, ?, ?)", 
                       (self.wx_id, "assistant", reply, time.time()))

            # 6. 线索回填与状态推进 (乐观锁)
            if extracted: bus.update_leads.emit(extracted)
            if new_status:
                lead = db.fetchone("SELECT version FROM crm_leads WHERE id=?", (self.wx_id,))
                if lead:
                    db.execute("UPDATE crm_leads SET status=?, version=version+1 WHERE id=? AND version=?", 
                               (new_status, self.wx_id, lead['version']))
            
            bus.status_update.emit("AI 就绪", "#a6e3a1")
        except Exception as e:
            bus.log_msg.emit(f"[Fatal] {e}", "danger")
            bus.status_update.emit("系统异常", "#f38ba8")
        finally:
            # 兜底防卡死
            pass

        leads = db.fetchall("SELECT * FROM crm_leads")
        self.table.setRowCount(0)
        for r in leads:
            row = self.table.rowCount()
            self.table.insertRow(row)
            self.table.setItem(row, 0, QTableWidgetItem(r["name"]))
            self.table.setItem(row, 1, QTableWidgetItem(r["intent"]))
            self.table.setItem(row, 2, QTableWidgetItem(str(r["value"])))
            self.table.setItem(row, 3, QTableWidgetItem(r["status"]))
            self.table.setItem(row, 4, QTableWidgetItem(str(r["version"])))

    def _test_lock(self):
        # 模拟线程1抢先更新
        lead = db.fetchone("SELECT version FROM crm_leads WHERE id='u001'")
        db.execute("UPDATE crm_leads SET status='已成交', version=version+1 WHERE id='u001' AND version=?", (lead['version'],))
        # 模拟线程2用旧version更新，应该失败
        cur = db.execute("UPDATE crm_leads SET status='战败', version=version+1 WHERE id='u001' AND version=?", (lead['version'],))
        if cur.rowcount == 0:
            bus.log_msg.emit("[CRM] 乐观锁生效：检测到并发冲突，拒绝覆盖！", "warning")
        self._refresh()
