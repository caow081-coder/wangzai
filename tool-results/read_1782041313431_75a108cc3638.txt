   848→   848→function timeAgo(iso: string): string {
   849→   849→  const diff = Date.now() - new Date(iso).getTime()
   850→   850→  if (diff < 60000) return '刚刚'
   851→   851→  if (diff < 3600000) return `${Math.floor(diff / 60000)}m`
   852→   852→  if (diff < 86400000) return `${Math.floor(diff / 3600000)}h`
   853→   853→  return `${Math.floor(diff / 86400000)}d`
   854→   854→}
